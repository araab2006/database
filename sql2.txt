CREATE OR REPLACE TYPE borrower_address_type AS OBJECT (
    street VARCHAR2(25),
    city VARCHAR2(25),
    country VARCHAR2(25)
);
/

CREATE OR REPLACE TYPE library_room_type AS OBJECT (
    room_number NUMBER,
    capacity NUMBER,
    description VARCHAR2(50)
);
/

CREATE OR REPLACE TYPE library_room_varray_type AS VARRAY(10) OF library_room_type;
/

CREATE OR REPLACE TYPE library_card_address_type AS OBJECT (
    street VARCHAR2(25),
    city VARCHAR2(25),
    country VARCHAR2(25)
);
/

CREATE TABLE card_addresses OF library_card_address_type;

CREATE TABLE borrowers (
    borrower_id NUMBER(6),
    borrower_name VARCHAR2(25),
    library_card_address REF library_card_address_type SCOPE IS card_addresses
);

CREATE OR REPLACE TYPE region_type AS OBJECT (
    region VARCHAR2(25),
    country VARCHAR2(25)
);
/

CREATE TABLE regions OF region_type;

ALTER TABLE library_branches ADD (region_ref REF region_type SCOPE IS regions);

CREATE OR REPLACE TYPE social_media_type AS OBJECT (
    media_name VARCHAR2(25),
    contact VARCHAR2(50)
);
/

CREATE OR REPLACE TYPE social_media_table_type AS TABLE OF social_media_type;
/

CREATE TABLE adverts (
    advert_id NUMBER(6),
    advert_title VARCHAR2(50),
    description VARCHAR2(25),
    social_media_used social_media_table_type
)
NESTED TABLE social_media_used STORE AS social_media_store_tab;

    social_media_table_type(
        social_media_type('FACEBOOK', '@readingclub'),
        social_media_type('TWITTER', '#books')
    ));

FROM adverts a, TABLE(a.social_media_used) s;
