SELECT m.member_id, m.member_firstname, p.publisher_name
FROM members m
INNER JOIN publishers p ON m.publisher_id = p.publisher_id;

SELECT bo.book_offering_id, b.book_name, l.librarian_name
FROM book_offerings bo
INNER JOIN books b ON bo.book_id = b.book_id
INNER JOIN librarians l ON bo.librarian_id = l.librarian_id;

SELECT m.member_firstname, b.book_name, bo.start_date
FROM members m
INNER JOIN loans l ON m.member_id = l.member_id
INNER JOIN book_offerings bo ON l.book_offering_id = bo.book_offering_id
INNER JOIN books b ON bo.book_id = b.book_id;

SELECT l.librarian_name, b.book_name, lb.location
FROM librarians l
INNER JOIN book_offerings bo ON l.librarian_id = bo.librarian_id
INNER JOIN books b ON bo.book_id = b.book_id
INNER JOIN library_branches lb ON bo.branch_id = lb.branch_id;

SELECT m.member_firstname, b.book_name
FROM members m, books b;

FROM librarians l, book_offerings bo;

SELECT m.member_id, m.member_firstname, lo.book_offering_id
FROM loans lo

SELECT l.librarian_name, bo.book_offering_id
