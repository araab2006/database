CREATE TABLE personal (
    personal_id    NUMBER(5) PRIMARY KEY,
    first_name     VARCHAR(7),
    last_name      VARCHAR(10),
    dob            DATE,
    nationality    VARCHAR(6),
    blood_group    VARCHAR(8)
);

CREATE TABLE address (
    personal_id       NUMBER(10),
    permanent_address VARCHAR(23),
    current_address   VARCHAR(23),
    address_type      VARCHAR(20),
    CONSTRAINT fk_address_personal FOREIGN KEY (personal_id)
        REFERENCES personal(personal_id)
);

CREATE TABLE contact (
    personal_id       NUMBER(18),
    primary_contact   NUMBER(11),
    secondary_contact NUMBER(15),
    home_contact      NUMBER(17),
    CONSTRAINT fk_contact_personal FOREIGN KEY (personal_id)
        REFERENCES personal(personal_id)
);

CREATE TABLE education (
    personal_id     NUMBER(9),
    institute_name  VARCHAR(15),
    field_of_study  VARCHAR(15),
    degree          VARCHAR(20),
    start_date      DATE,
    end_date        DATE,
    grade           VARCHAR(3),
    CONSTRAINT fk_education_personal FOREIGN KEY (personal_id)
        REFERENCES personal(personal_id)
);

CREATE TABLE medical_records (
    personal_id   NUMBER(8),
    condition_name VARCHAR(18),
    treatment     VARCHAR(17),
    doctor_name   VARCHAR(19),
    hospital_name VARCHAR(20),
    CONSTRAINT fk_medical_personal FOREIGN KEY (personal_id)
        REFERENCES personal(personal_id)
);

CREATE TABLE document (
    personal_id     NUMBER(9),
    document_type   VARCHAR(15),
    issued_date     DATE,
    expiry_date     DATE,
    document_number NUMBER(10),
    CONSTRAINT fk_document_personal FOREIGN KEY (personal_id)
        REFERENCES personal(personal_id)
);
